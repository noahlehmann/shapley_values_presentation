    \section{Introduction}
    \begin{frame}{Introduction to Emotion Analysis}
        Why?
        \begin{itemize}
            \item Recognize major changes in behaviour
            \item Pre-classify hate speech
            \item Detect depressive tendencies
        \end{itemize}
        How?
        \begin{itemize}
            \item Well-balanced data from multiple cultures
            \item Detect tendencies of word combinations
            \item Use on social media, text messengers and mail services
        \end{itemize}
    \end{frame}

    \begin{frame}{Text Classification}
        What data-set is used?
        \textrightarrow{} Text collection with five emotions (anger, fear, sadness, neutral, happy)
        \begin{enumerate}
            \item 5 emotions \textrightarrow{} further insights \linebreak higher complexity
            \item Only detect sadness \textrightarrow{} specialized \linebreak less data required
        \end{enumerate}
    \end{frame}

    \section{Solution}

    \begin{frame}{Solution}{Text Preparation}
        \begin{itemize}
            \item Reduce Data-set to binary classification
            \item Sample all non-sad-classified rows
            \item Remove Stopwords
            \item Reduce word complixity
            \item Tokenize Texts
        \end{itemize}
        Primary Goal: Reduce complexity and feature set
    \end{frame}

    \begin{frame}{Solution}{Classification}
        Attempt and tweak multiple classifiers
        \begin{itemize}
            \item Naive Bayes
            \item Random Forest
            \item Neural Network
        \end{itemize}
        \textcolor{red}{How do we get insights into the more complex models?}
        \begin{itemize}
            \item Some have good explainability
            \item For others: \textbf{Shapley Values}
        \end{itemize}
    \end{frame}
